pipeline {
    agent any 

    stages {
        stage('Cleaning Workspace') {
            steps {
                cleanWs()
            }
        }
        
        stage('Checkout from Git') {
            steps {
                git branch: 'main', url: 'https://github.com/sonalrmeshram/fullstack-authors-books-application.git'
            }
        }
        
        stage("Docker Image Build") {
            steps {
                script {
                    dir('frontend') {
                        sh 'docker system prune -f'
                        sh 'docker container prune -f'
                        sh 'docker build -t frontend .'
                    }
                }
            }
        }
        
        stage("ECR Image Pushing") {
            steps {
                script {
                    sh '''
                    aws ecr get-login-password --region us-east-1 | docker login --username AWS --password-stdin 157664038953.dkr.ecr.us-east-1.amazonaws.com
                    docker tag frontend:latest 157664038953.dkr.ecr.us-east-1.amazonaws.com/frontend:${BUILD_NUMBER}
                    docker push 157664038953.dkr.ecr.us-east-1.amazonaws.com/frontend:${BUILD_NUMBER}
                    docker rmi 157664038953.dkr.ecr.us-east-1.amazonaws.com/frontend:${BUILD_NUMBER}
                    '''
                }
            }
        }
        
        stage('Update Deployment file') {
            environment {
                GIT_REPO_NAME = "fullstack-authors-books-application"
                GIT_USER_NAME = "sonalrmeshram"
            }
            steps {
                withCredentials([string(credentialsId: 'my-git-pattoken', variable: 'git_token')]) {
                    sh '''
                        set -e
                        git config user.email "sonalrmeshram@gmail.com"
                        git config user.name "sonalrmeshram"
                        BUILD_NUMBER=${BUILD_NUMBER}
                        echo "Build Number: $BUILD_NUMBER"
                        
                        # Update deployment file with new image
                        sed -i "s#image:.*#image: 157664038953.dkr.ecr.us-east-1.amazonaws.com/frontend:$BUILD_NUMBER#g" argocd/frontend/argocd-frontend.yml
                        
                        # Commit and push changes
                        git add argocd/frontend/argocd-frontend.yml
                        git commit -m "Update frontend deployment image to version $BUILD_NUMBER"
                        git push https://${git_token}@github.com/${GIT_USER_NAME}/${GIT_REPO_NAME} HEAD:main
                    '''
                }
            }
        }
    }
}
